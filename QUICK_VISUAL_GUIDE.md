## Code Pattern to Follow

```typescript
// Pattern for opening Media Library

const [showMediaLibrary, setShowMediaLibrary] = useState(false);
const [selectedMedia, setSelectedMedia] = useState<string[]>([]);

// Button to open Media Library
<button onClick={() => setShowMediaLibrary(true)}>
  рҹ“Ғ Select from Media Library
</button>

// Media Library Modal
{showMediaLibrary && (
  <MediaLibraryModal
    mode="multiple" // or "single"
    onSelect={(urls) => {
      setSelectedMedia(urls);
      setShowMediaLibrary(false);
    }}
    onClose={() => setShowMediaLibrary(false)}
  />
)}

// Display selected media
{selectedMedia.map(url => (
  <img key={url} src={url} alt="Selected" />
))}
```
